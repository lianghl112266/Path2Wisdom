```
semaphore empty = N
semaphore mutex = 1
semaphore odd = 0
semaphore even = 0

p1 {
    while(1) {
        x = produce()
        p(empty)
        p(mutex)
        put()
        v(mutex)
        if (x%2 == 1) v(odd)
        else v(even)
    }
}

p2 {
    while(1) {
        p(odd)
        p(mutex)
        getodd()
        v(mutex)
        v(empty)
        countodd()
    }
}

p2 {
    while(1) {
        p(even)
        p(mutex)
        geteven()
        v(mutex)
        v(empty)
        counteven()
    }
}
```
